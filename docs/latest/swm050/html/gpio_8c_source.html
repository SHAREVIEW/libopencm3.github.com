28 /** @brief Set a Group of Pins
29 
30 Set one or more pins of GPIO to 1. Please note that this chip doesn't support
31 atomic pin setting.
32 
33 @param[in] gpios Pin identifiers @ref gpio_pin_id
34  If multiple pins are to be changed, use bitwise OR '|' to separate

72 /** @brief Toggle a Group of Pins
73 
74 Toggle one or more pins of GPIO. The non-toggled pins are not affected.
75 
76 @param[in] gpios Pin identifiers @ref gpio_pin_id
77  If multiple pins are to be changed, use bitwise OR '|' to separate
78  them.
79 */
80 void gpio_toggle(uint16_t gpios)
81 {
82  uint32_t curr_status = GPIO_DATA & gpios;
83  GPIO_DATA = (GPIO_DATA & (~gpios)) | (~curr_status);
84 }

115 /** @brief Select the alternative function of a Group of Pins
116 
117 Select the alternative function of one or more pins of GPIO.
118 
119 @param[in] gpios Pin identifiers @ref gpio_pin_id
120  If multiple pins are to be changed, use bitwise OR '|' to separate
121  them.
122 @param[in] af_en Whether alternative function is selected
123 */
124 void gpio_sel_af(uint16_t gpios, bool af_en)
125 {
126  if (gpios & GPIO0) {
127  GPIO_SEL = (GPIO_SEL & (~0x3)) | (af_en ? 0x1 : 0x0);
128  }
129  if (gpios & GPIO1) {
130  GPIO_SEL = (GPIO_SEL & (~0xc)) | (af_en ? 0x4 : 0x0);
131  }
132  if (gpios & GPIO2) {
133  GPIO_SEL = (GPIO_SEL & (~0x30)) | (af_en ? 0x10 : 0x0);
134  }
135  if (gpios & GPIO7) {
136  GPIO_SEL = (GPIO_SEL & (~0xc000)) | (af_en ? 0x4000 : 0x0);
137  }
138 }

143 Enable or disable the internal pull-up of one or more pins of GPIO.

145 @param[in] gpios Pin identifiers @ref gpio_pin_id

163 the input function of pins decreases the power usage of the MCU.
164 
165 @param[in] gpios Pin identifiers @ref gpio_pin_id
166  If multiple pins are to be changed, use bitwise OR '|' to separate

170 void gpio_in_en(uint16_t gpios, bool en)
171 {
172  if (en) {
173  GPIO_INEN &= ~gpios;
174  } else {
175  GPIO_INEN |= gpios;
176  }

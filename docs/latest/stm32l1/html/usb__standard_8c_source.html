68 static uint16_t build_config_descriptor(usbd_device *usbd_dev,
69  uint8_t index, uint8_t *buf, uint16_t len)
70 {
71  uint8_t *tmpbuf = buf;
72  const struct usb_config_descriptor *cfg = &usbd_dev->config[index];
73  uint16_t count, total = 0, totallen = 0;
74  uint16_t i, j, k;
75 
76  memcpy(buf, cfg, count = MIN(len, cfg->bLength));
77  buf += count;
78  len -= count;
79  total += count;
80  totallen += cfg->bLength;
81 
82  /* For each interface... */
83  for (i = 0; i < cfg->bNumInterfaces; i++) {
84  /* Interface Association Descriptor, if any */
85  if (cfg->interface[i].iface_assoc) {
86  const struct usb_iface_assoc_descriptor *assoc =
87  cfg->interface[i].iface_assoc;
88  memcpy(buf, assoc, count = MIN(len, assoc->bLength));
89  buf += count;
90  len -= count;
91  total += count;
92  totallen += assoc->bLength;
93  }
94  /* For each alternate setting... */
95  for (j = 0; j < cfg->interface[i].num_altsetting; j++) {
96  const struct usb_interface_descriptor *iface =
97  &cfg->interface[i].altsetting[j];
98  /* Copy interface descriptor. */
99  memcpy(buf, iface, count = MIN(len, iface->bLength));
100  buf += count;
101  len -= count;
102  total += count;
103  totallen += iface->bLength;
104  /* Copy extra bytes (function descriptors). */
105  if (iface->extra) {
106  memcpy(buf, iface->extra,
107  count = MIN(len, iface->extralen));
108  buf += count;
109  len -= count;
110  total += count;
111  totallen += iface->extralen;
112  }
113  /* For each endpoint... */
114  for (k = 0; k < iface->bNumEndpoints; k++) {
115  const struct usb_endpoint_descriptor *ep =
116  &iface->endpoint[k];
117  memcpy(buf, ep, count = MIN(len, ep->bLength));
118  buf += count;
119  len -= count;
120  total += count;
121  totallen += ep->bLength;
122  /* Copy extra bytes (class specific). */
123  if (ep->extra) {
124  memcpy(buf, ep->extra,
125  count = MIN(len, ep->extralen));
126  buf += count;
127  len -= count;
128  total += count;
129  totallen += ep->extralen;
130  }
131  }
132  }
133  }
134 
135  /* Fill in wTotalLength.
136  * Note that tmpbuf is sometimes not halfword-aligned */
137  memcpy((tmpbuf + 2), &totallen, sizeof(uint16_t));
138 
139  return total;
140 }

152 static enum usbd_request_return_codes
153 usb_standard_get_descriptor(usbd_device *usbd_dev,
154  struct usb_setup_data *req,
155  uint8_t **buf, uint16_t *len)
156 {
157  int i, array_idx, descr_idx;
158  struct usb_string_descriptor *sd;
159 
160  descr_idx = usb_descriptor_index(req->wValue);
161 
162  switch (usb_descriptor_type(req->wValue)) {
163  case USB_DT_DEVICE:
164  *buf = (uint8_t *) usbd_dev->desc;
165  *len = MIN(*len, usbd_dev->desc->bLength);
166  return USBD_REQ_HANDLED;
167  case USB_DT_CONFIGURATION:
168  *buf = usbd_dev->ctrl_buf;
169  *len = build_config_descriptor(usbd_dev, descr_idx, *buf, *len);
170  return USBD_REQ_HANDLED;
171  case USB_DT_STRING:
172  sd = (struct usb_string_descriptor *)usbd_dev->ctrl_buf;
173 
174  if (descr_idx == 0) {
175  /* Send sane Language ID descriptor... */
176  sd->wData[0] = USB_LANGID_ENGLISH_US;
177  sd->bLength = sizeof(sd->bLength) +
178  sizeof(sd->bDescriptorType) +
179  sizeof(sd->wData[0]);
180 
181  *len = MIN(*len, sd->bLength);
182  } else {
183  array_idx = descr_idx - 1;
184 
185  if (!usbd_dev->strings) {
186  /* Device doesn't support strings. */
187  return USBD_REQ_NOTSUPP;
188  }
189 
190  /* Check that string index is in range. */
191  if (array_idx >= usbd_dev->num_strings) {
192  return USBD_REQ_NOTSUPP;
193  }
194 
195  /* Strings with Language ID differnet from
196  * USB_LANGID_ENGLISH_US are not supported */
197  if (req->wIndex != USB_LANGID_ENGLISH_US) {
198  return USBD_REQ_NOTSUPP;
199  }
200 
201  /* This string is returned as UTF16, hence the
202  * multiplication
203  */
204  sd->bLength = strlen(usbd_dev->strings[array_idx]) * 2 +
205  sizeof(sd->bLength) +
206  sizeof(sd->bDescriptorType);
207 
208  *len = MIN(*len, sd->bLength);
209 
210  for (i = 0; i < (*len / 2) - 1; i++) {
211  sd->wData[i] =
212  usbd_dev->strings[array_idx][i];
213  }
214  }
215 
216  sd->bDescriptorType = USB_DT_STRING;
217  *buf = (uint8_t *)sd;
218 
219  return USBD_REQ_HANDLED;
220  }
221  return USBD_REQ_NOTSUPP;
222 }

251 static enum usbd_request_return_codes
252 usb_standard_set_configuration(usbd_device *usbd_dev,
253  struct usb_setup_data *req,
254  uint8_t **buf, uint16_t *len)
255 {
256  unsigned i;
257  int found_index = -1;
258  const struct usb_config_descriptor *cfg;
259 
260  (void)req;
261  (void)buf;
262  (void)len;
263 
264  if (req->wValue > 0) {
265  for (i = 0; i < usbd_dev->desc->bNumConfigurations; i++) {
266  if (req->wValue
267  == usbd_dev->config[i].bConfigurationValue) {
268  found_index = i;
269  break;
270  }
271  }
272  if (found_index < 0) {
273  return USBD_REQ_NOTSUPP;
274  }
275  }
276 
277  usbd_dev->current_config = found_index + 1;
278 
279  if (usbd_dev->current_config > 0) {
280  cfg = &usbd_dev->config[usbd_dev->current_config - 1];
281 
282  /* reset all alternate settings configuration */
283  for (i = 0; i < cfg->bNumInterfaces; i++) {
284  if (cfg->interface[i].cur_altsetting) {
285  *cfg->interface[i].cur_altsetting = 0;
286  }
287  }
288  }
289 
290  /* Reset all endpoints. */
291  usbd_dev->driver->ep_reset(usbd_dev);
292 
293  if (usbd_dev->user_callback_set_config[0]) {
294  /*
295  * Flush control callbacks. These will be reregistered
296  * by the user handler.
297  */
298  for (i = 0; i < MAX_USER_CONTROL_CALLBACK; i++) {
299  usbd_dev->user_control_callback[i].cb = NULL;
300  }
301 
302  for (i = 0; i < MAX_USER_SET_CONFIG_CALLBACK; i++) {
303  if (usbd_dev->user_callback_set_config[i]) {
304  usbd_dev->user_callback_set_config[i](usbd_dev,
305  req->wValue);
306  }
307  }
308  }
309 
310  return USBD_REQ_HANDLED;
311 }

474 enum usbd_request_return_codes
475 _usbd_standard_request_device(usbd_device *usbd_dev,
476  struct usb_setup_data *req, uint8_t **buf,
477  uint16_t *len)
478 {
479  enum usbd_request_return_codes (*command)(usbd_device *usbd_dev,
480  struct usb_setup_data *req,
481  uint8_t **buf, uint16_t *len) = NULL;
482 
483  switch (req->bRequest) {
484  case USB_REQ_CLEAR_FEATURE:
485  case USB_REQ_SET_FEATURE:
486  if (req->wValue == USB_FEAT_DEVICE_REMOTE_WAKEUP) {
487  /* Device wakeup code goes here. */
488  }
489 
490  if (req->wValue == USB_FEAT_TEST_MODE) {
491  /* Test mode code goes here. */
492  }
493 
494  break;
495  case USB_REQ_SET_ADDRESS:
496  /*
497  * SET ADDRESS is an exception.
498  * It is only processed at STATUS stage.
499  */
500  command = usb_standard_set_address;
501  break;
502  case USB_REQ_SET_CONFIGURATION:
503  command = usb_standard_set_configuration;
504  break;
505  case USB_REQ_GET_CONFIGURATION:
506  command = usb_standard_get_configuration;
507  break;
508  case USB_REQ_GET_DESCRIPTOR:
509  command = usb_standard_get_descriptor;
510  break;
511  case USB_REQ_GET_STATUS:
512  /*
513  * GET_STATUS always responds with zero reply.
514  * The application may override this behaviour.
515  */
516  command = usb_standard_device_get_status;
517  break;
518  case USB_REQ_SET_DESCRIPTOR:
519  /* SET_DESCRIPTOR is optional and not implemented. */
520  break;
521  }
522 
523  if (!command) {
524  return USBD_REQ_NOTSUPP;
525  }
526 
527  return command(usbd_dev, req, buf, len);
528 }

601 enum usbd_request_return_codes
602 _usbd_standard_request(usbd_device *usbd_dev, struct usb_setup_data *req,
603  uint8_t **buf, uint16_t *len)
604 {
605  /* FIXME: Have class/vendor requests as well. */
606  if ((req->bmRequestType & USB_REQ_TYPE_TYPE) != USB_REQ_TYPE_STANDARD) {
607  return USBD_REQ_NOTSUPP;
608  }
609 
610  switch (req->bmRequestType & USB_REQ_TYPE_RECIPIENT) {
611  case USB_REQ_TYPE_DEVICE:
612  return _usbd_standard_request_device(usbd_dev, req, buf, len);
613  case USB_REQ_TYPE_INTERFACE:
614  return _usbd_standard_request_interface(usbd_dev, req,
615  buf, len);
616  case USB_REQ_TYPE_ENDPOINT:
617  return _usbd_standard_request_endpoint(usbd_dev, req, buf, len);
618  default:
619  return USBD_REQ_NOTSUPP;
620  }
621 }

100 /** @brief Write a word to eeprom
101  *
102  * @param address assumed to be in the eeprom space, no checking
103  * @param data word to write
104  */
105 void eeprom_program_word(uint32_t address, uint32_t data)
106 {
107  flash_unlock_pecr();
108  /* erase only if needed */
109  FLASH_PECR &= ~FLASH_PECR_FTDW;
110  MMIO32(address) = data;
111  flash_lock_pecr();
112 }

114 /** @brief Write a block of words to eeprom
115  *
116  * Writes a block of words to EEPROM at the requested address, erasing if
117  * necessary, and locking afterwards. Only wordwise writing is safe for
118  * writing any value
119  *
120  * @param[in] address must point to EEPROM space, no checking!
121  * @param[in] data pointer to data to write
122  * @param[in] length_in_words size of of data in WORDS!
123  */
124 void eeprom_program_words(uint32_t address, uint32_t *data, int length_in_words)
125 {
126  int i;
127  flash_unlock_pecr();
128  while (FLASH_SR & FLASH_SR_BSY);
129  /* erase only if needed */
130  FLASH_PECR &= ~FLASH_PECR_FTDW;
131  for (i = 0; i < length_in_words; i++) {
132  MMIO32(address + (i * sizeof(uint32_t))) = *(data+i);
133  while (FLASH_SR & FLASH_SR_BSY);
134  }
135  flash_lock_pecr();
136 }

34 void desig_get_unique_id_as_string(char *string,
35  unsigned int string_len)
36 {
37  int i, len;
38  uint32_t dev_id_buf[3];
39  uint8_t *device_id = (uint8_t *)dev_id_buf;
40  const char chars[] = "0123456789ABCDEF";
41 
42  desig_get_unique_id(dev_id_buf);
43 
44  /* Each byte produces two characters */
45  len = (2 * sizeof(dev_id_buf) < string_len) ?
46  2 * sizeof(dev_id_buf) : string_len - 1;
47 
48  for (i = 0; i < len; i += 2) {
49  string[i] = chars[(device_id[i / 2] >> 4) & 0x0F];
50  string[i + 1] = chars[(device_id[i / 2] >> 0) & 0x0F];
51  }

56 /**
57  * Generate a serial number from the unique id registers as the DFU bootloader.
58  * This document: http://www.usb.org/developers/docs/devclass_docs/usbmassbulk_10.pdf
59  * says that the serial number has to be at least 12 digits long and that
60  * the last 12 digits need to be unique. It also stipulates that the valid
61  * character set is that of upper-case hexadecimal digits.
62  * The onboard DFU bootloader produces a 12-digit serial based on the
63  * 96-bit unique ID. Show the serial with ```dfu-util -l``` while the
64  * MCU is in DFU mode.
65  * @see https://my.st.com/52d187b7 for the algorithim used.
66  * @param string pointer to store serial in, must be at least 13 bytes
67  */
68 void desig_get_unique_id_as_dfu(char *string) {
69  uint8_t *id = (uint8_t *)DESIG_UNIQUE_ID_BASE;
70 
71  uint8_t serial[6];
72  serial[0] = id[11];
73  serial[1] = id[10] + id[2];
74  serial[2] = id[9];
75  serial[3] = id[8] + id[0];
76  serial[4] = id[7];
77  serial[5] = id[6];
78 
79  uint8_t *ser = &serial[0];
80  uint8_t *end = &serial[6];
81  const char hex_digit[] = "0123456789ABCDEF";
82 
83  for (; ser < end; ser++) {
84  *string++ = hex_digit[(*ser >> 4) & 0x0f];
85  *string++ = hex_digit[(*ser >> 0) & 0x0f];
86  }
87  *string = '\0';
88 }

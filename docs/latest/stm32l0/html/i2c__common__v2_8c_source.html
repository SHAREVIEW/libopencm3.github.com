28 /** @brief I2C Reset.
29  *
30  * The I2C peripheral and all its associated configuration registers are placed
31  * in the reset condition. The reset is effected via the RCC peripheral reset
32  * system.
33  *
34  * @param[in] i2c Unsigned int32. I2C peripheral identifier @ref i2c_reg_base.
35  */
36 
37 void i2c_reset(uint32_t i2c)
38 {
39  switch (i2c) {
40  case I2C1:
41  rcc_periph_reset_pulse(RST_I2C1);
42  break;
43 #if defined(I2C2_BASE)
44  case I2C2:
45  rcc_periph_reset_pulse(RST_I2C2);
46  break;
47 #endif
48 #if defined(I2C3_BASE)
49  case I2C3:
50  rcc_periph_reset_pulse(RST_I2C3);
51  break;
52 #endif
53 #if defined(I2C4_BASE)
54  case I2C4:
55  rcc_periph_reset_pulse(RST_I2C4);
56  break;
57 #endif
58  default:
59  break;
60  }
61 }

193  * These bits are used to configure the digital noise filter on SDA and

387 /**
388  * Run a write/read transaction to a given 7bit i2c address
389  * If both write & read are provided, the read will use repeated start.
390  * Both write and read are optional
391  * @param i2c peripheral of choice, eg I2C1
392  * @param addr 7 bit i2c device address
393  * @param w buffer of data to write
394  * @param wn length of w
395  * @param r destination buffer to read into
396  * @param rn number of bytes to read (r should be at least this long)
397  */
398 void i2c_transfer7(uint32_t i2c, uint8_t addr, uint8_t *w, size_t wn, uint8_t *r, size_t rn)
399 {
400  /* waiting for busy is unnecessary. read the RM */
401  if (wn) {
402  i2c_set_7bit_address(i2c, addr);
403  i2c_set_write_transfer_dir(i2c);
404  i2c_set_bytes_to_transfer(i2c, wn);
405  if (rn) {
406  i2c_disable_autoend(i2c);
407  } else {
408  i2c_enable_autoend(i2c);
409  }
410  i2c_send_start(i2c);
411 
412  while (wn--) {
413  bool wait = true;
414  while (wait) {
415  if (i2c_transmit_int_status(i2c)) {
416  wait = false;
417  }
418  while (i2c_nack(i2c)); /* FIXME Some error */
419  }
420  i2c_send_data(i2c, *w++);
421  }
422  /* not entirely sure this is really necessary.
423  * RM implies it will stall until it can write out the later bits
424  */
425  if (rn) {
426  while (!i2c_transfer_complete(i2c));
427  }
428  }
429 
430  if (rn) {
431  /* Setting transfer properties */
432  i2c_set_7bit_address(i2c, addr);
433  i2c_set_read_transfer_dir(i2c);
434  i2c_set_bytes_to_transfer(i2c, rn);
435  /* start transfer */
436  i2c_send_start(i2c);
437  /* important to do it afterwards to do a proper repeated start! */
438  i2c_enable_autoend(i2c);
439 
440  for (size_t i = 0; i < rn; i++) {
441  while (i2c_received_data(i2c) == 0);
442  r[i] = i2c_get_data(i2c);
443  }
444  }
445 }

448 /**
449  * Set the i2c communication speed.
450  * NOTE: 1MHz mode not yet implemented!
451  * Min clock speed: 8MHz for FM, 2Mhz for SM,
452  * @param i2c peripheral, eg I2C1
453  * @param speed one of the listed speed modes @ref i2c_speeds
454  * @param clock_megahz i2c peripheral clock speed in MHz. Usually, rcc_apb1_frequency / 1e6
455  */
456 void i2c_set_speed(uint32_t i2c, enum i2c_speeds speed, uint32_t clock_megahz)
457 {
458  int prescaler;
459  switch(speed) {
460  case i2c_speed_fmp_1m:
461  /* FIXME - add support for this mode! */
462  break;
463  case i2c_speed_fm_400k:
464  /* target 8Mhz input, so tpresc = 125ns */
465  prescaler = clock_megahz / 8 - 1;
466  i2c_set_prescaler(i2c, prescaler);
467  i2c_set_scl_low_period(i2c, 10-1); // 1250ns
468  i2c_set_scl_high_period(i2c, 4-1); // 500ns
469  i2c_set_data_hold_time(i2c, 3); // 375ns
470  i2c_set_data_setup_time(i2c, 4-1); // 500ns
471  break;
472  default:
473  /* fall back to standard mode */
474  case i2c_speed_sm_100k:
475  /* target 4Mhz input, so tpresc = 250ns */
476  prescaler = (clock_megahz / 4) - 1;
477  i2c_set_prescaler(i2c, prescaler);
478  i2c_set_scl_low_period(i2c, 20-1); // 5usecs
479  i2c_set_scl_high_period(i2c, 16-1); // 4usecs
480  i2c_set_data_hold_time(i2c, 2); // 0.5usecs
481  i2c_set_data_setup_time(i2c, 5-1); // 1.25usecs
482  break;
483  }
484 }

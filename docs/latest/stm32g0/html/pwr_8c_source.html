82 void pwr_enable_power_voltage_detect(uint32_t pvdr_level, uint32_t pvdf_level)
83 {

85 
86  reg32 = PWR_CR2;

88  reg32 &= ~(PWR_CR2_PVDFT_MASK << PWR_CR2_PVDFT_SHIFT);
89  PWR_CR2 = (reg32 | (pvdf_level << PWR_CR2_PVDFT_SHIFT) | (pvdr_level << PWR_CR2_PVDRT_SHIFT) | PWR_CR2_PVDE);

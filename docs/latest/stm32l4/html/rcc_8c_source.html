192 void rcc_wait_for_sysclk_status(enum rcc_osc osc)
193 {
194  switch (osc) {
195  case RCC_PLL:
196  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
197  != RCC_CFGR_SWS_PLL);
198  break;
199  case RCC_HSE:
200  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
201  != RCC_CFGR_SWS_HSE);
202  break;
203  case RCC_HSI16:
204  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
205  != RCC_CFGR_SWS_HSI16);
206  break;
207  case RCC_MSI:
208  while (((RCC_CFGR >> RCC_CFGR_SWS_SHIFT) & RCC_CFGR_SWS_MASK)
209  != RCC_CFGR_SWS_MSI);
210  break;
211  default:
212  /* Shouldn't be reached. */
213  break;
214  }
215 }

343 /**
344  * Set the msi run time range.
345  * Can only be called when MSI is either OFF, or when MSI is on _and_
346  * ready. (RCC_CR_MSIRDY bit). @sa rcc_set_msi_range_standby
347  * @param msi_range range number @ref rcc_cr_msirange
348  */
349 void rcc_set_msi_range(uint32_t msi_range)
350 {
351  uint32_t reg = RCC_CR;
352  reg &= ~(RCC_CR_MSIRANGE_MASK << RCC_CR_MSIRANGE_SHIFT);
353  reg |= msi_range << RCC_CR_MSIRANGE_SHIFT;
354  RCC_CR = reg | RCC_CR_MSIRGSEL;
355 }

360  * Note that not all MSI range values are allowed here!

362  * @param msi_range range number valid for post standby @ref rcc_csr_msirange

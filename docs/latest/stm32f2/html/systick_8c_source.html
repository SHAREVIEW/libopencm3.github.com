44 /** @brief SysTick Set the Automatic Reload Value.
45  *
46  * The counter is set to the reload value when the counter starts and after it
47  * reaches zero.
48  *
49  * @note The systick counter value might be undefined upon startup. To get
50  * predictable behavior, it is a good idea to set or clear the counter after
51  * set reload. @sa systick_clear
52  *
53  * @param[in] value uint32_t. 24 bit reload value.
54  */
55 
56 void systick_set_reload(uint32_t value)
57 {
58  STK_RVR = (value & STK_RVR_RELOAD);
59 }
60

62 /** @brief SysTick Read the Automatic Reload Value.
63  *
64  * @returns 24 bit reload value as uint32_t.
65  */
66 
67 uint32_t systick_get_reload(void)
68 {
69  return STK_RVR & STK_RVR_RELOAD;
70 }

72 /** @brief SysTick Set clock and frequency of overflow
73  *
74  * This function sets the systick to AHB clock source, and the prescaler to
75  * generate interrupts with the desired frequency. The function fails, if
76  * the frequency is too low.
77  *
78  * @param[in] freq uint32_t The desired frequency in Hz
79  * @param[in] ahb uint32_t The current AHB frequency in Hz
80  * @returns true, if success, false if the desired frequency cannot be set.
81  */
82 bool systick_set_frequency(uint32_t freq, uint32_t ahb)
83 {
84  uint32_t ratio = ahb / freq;
85 
86 #if defined(__ARM_ARCH_6M__)
87  systick_set_clocksource(STK_CSR_CLKSOURCE_AHB);
88 #else
89  if (ratio >= (STK_RVR_RELOAD * 8)) {
90  /* This frequency is too slow */
91  return false;
92  } else if (ratio >= STK_RVR_RELOAD) {
93  ratio /= 8;
94  systick_set_clocksource(STK_CSR_CLKSOURCE_AHB_DIV8);
95  } else {
96  systick_set_clocksource(STK_CSR_CLKSOURCE_AHB);
97  }
98 #endif
99  systick_set_reload(ratio - 1);
100  return true;
101 }

115 /** @brief Set the SysTick Clock Source.
116  *
117  * The clock source can be either the AHB clock or the same clock divided by 8.
118  *
119  * @param[in] clocksource uint8_t. Clock source from @ref systick_clksource.
120  */
121 
122 void systick_set_clocksource(uint8_t clocksource)
123 {
124  STK_CSR = (STK_CSR & ~STK_CSR_CLKSOURCE) |
125  (clocksource & STK_CSR_CLKSOURCE);
126 }

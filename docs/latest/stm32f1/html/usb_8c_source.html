97 void _usbd_reset(usbd_device *usbd_dev)
98 {
99  usbd_dev->current_address = 0;
100  usbd_dev->current_config = 0;
101  usbd_ep_setup(usbd_dev, 0, USB_ENDPOINT_ATTR_CONTROL, usbd_dev->desc->bMaxPacketSize0, NULL);
102  usbd_dev->driver->set_address(usbd_dev, 0);
103 
104  if (usbd_dev->user_callback_reset) {
105  usbd_dev->user_callback_reset();
106  }
107 }

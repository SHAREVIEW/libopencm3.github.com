32 /** @brief Issue Pipeline Stall
33 
34 Issue a pipeline stall to make sure all write operations completed.
35 
36 RM0385: After performing a data write operation and before polling the BSY bit
37 to be cleared, the software can issue a DSB instruction to guarantee the
38 completion of a previous data write operation.
39 
40 */
41 
42 static inline void flash_pipeline_stall(void)
43 {
44  __asm__ volatile("dsb":::"memory");
45 }
46 
47 /*---------------------------------------------------------------------------*/
48 /** @brief Wait until Last Operation has Ended
49 
50 This loops indefinitely until an operation (write or erase) has completed by
51 testing the busy flag.
52 */
53 void flash_wait_for_last_operation(void)
54 {
55  flash_pipeline_stall();
56  while ((FLASH_SR & FLASH_SR_BSY) == FLASH_SR_BSY);
57 }

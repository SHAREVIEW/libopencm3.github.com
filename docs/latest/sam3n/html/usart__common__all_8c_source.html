22 void usart_set_databits(uint32_t usart, int bits)
23 {
24  USART_MR(usart) = (USART_MR(usart) & ~USART_MR_CHRL_MASK) |
25  ((bits - 5) << 6);
26 }

28 void usart_set_stopbits(uint32_t usart, enum usart_stopbits sb)
29 {

39 void usart_set_mode(uint32_t usart, enum usart_mode mode)
40 {
41  USART_CR(usart) =
42  (mode & USART_MODE_RX) ? USART_CR_RXEN : USART_CR_RXDIS;
43  USART_CR(usart) = (mode & USART_MODE_TX) ? USART_CR_TXEN
44  : USART_CR_TXDIS;
45 }

47 void usart_set_flow_control(uint32_t usart, enum usart_flowcontrol fc)
48 {
49  USART_MR(usart) = (USART_MR(usart) & ~USART_MR_MODE_MASK) |
50  (fc ? USART_MR_MODE_HW_HANDSHAKING : 0);
51 }

73 void usart_wait_send_ready(uint32_t usart)
74 {
75  while ((USART_CSR(usart) & USART_CSR_TXRDY) == 0);

78 void usart_wait_recv_ready(uint32_t usart)

116 void usart_select_clock(uint32_t usart, enum usart_clock clk)
117 {
118  uint32_t reg_mr = USART_MR(usart) & (~USART_MR_USCLKS_MASK);
119  USART_MR(usart) = ((clk << USART_MR_USCLKS_SHIFT) & USART_MR_USCLKS_MASK) | reg_mr;
120 }

122 void usart_set_character_length(uint32_t usart, enum usart_chrl chrl)
123 {
124  uint32_t reg_mr = USART_MR(usart) & (~USART_MR_CHRL_MASK);
125  USART_MR(usart) = reg_mr | (chrl << USART_MR_CHRL_SHIFT);
126 }
